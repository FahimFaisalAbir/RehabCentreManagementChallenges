CREATE TABLE ADMIN(
  AD_EMAIL VARCHAR(30) NOT NULL ,
  AD_PASSWORD VARCHAR(15) UNIQUE NOT NULL ,
  AD_ID INTEGER PRIMARY KEY AUTO_INCREMENT
);
SHOW TABLES ;
DESCRIBE CLIENTS;
CREATE TABLE CLIENTS(
  FNAME VARCHAR(15) NOT NULL ,
  LNAME VARCHAR(15),
  EMAIL VARCHAR(30) ,
  PASSWORD VARCHAR(15) UNIQUE NOT NULL ,
  AMOUNT FLOAT,
  ADMIN_ID INTEGER,
  PRIMARY KEY (EMAIL),
  FOREIGN KEY (ADMIN_ID) REFERENCES admin(AD_ID)
  ON DELETE SET NULL ON UPDATE CASCADE
);

CREATE TABLE FOOD(
  NAME VARCHAR(30) NOT NULL,
  F_ID INTEGER NOT NULL,
  PRICE DOUBLE NOT NULL,
  ADMIN_ID INTEGER,
  PRIMARY KEY (NAME),
  FOREIGN KEY (ADMIN_ID) REFERENCES ADMIN(AD_ID)
    ON DELETE SET NULL ON UPDATE CASCADE
);
SHOW TABLES;
DESCRIBE ORDER_INFO;

CREATE TABLE ORDER_INFO(
  EMAIL VARCHAR(30),
  FNAME VARCHAR(30),
  ORDER_NO INTEGER UNIQUE NOT NULL,
  QUANTITY INTEGER NOT NULL,
  PRICE_TOTAL DOUBLE,
  FOREIGN KEY (EMAIL) REFERENCES clients(EMAIL) ON UPDATE CASCADE,
  FOREIGN KEY (FNAME) REFERENCES food(NAME) ON UPDATE CASCADE,
  PRIMARY KEY (EMAIL,FNAME)
);
